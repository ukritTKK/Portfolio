router.post('/register', function(req, res, next) {
  var tmpUser = new User(req.body), obj;
  // console.log(req.body);
  // console.log(tmpUser);
  tmpUser.username = req.body.username;
  tmpUser.password = req.body.password;
  tmpUser.user_type = 0;
  tmpUser.save(function(error)
  {
    // console.log("delta 0")
    if(!error)
    {
      // console.log("delta 1")
      obj = {"message":"sign up and generate link successfully"};
      res.status(200).json({obj})
      return 1;
    }
    else
    {
      // console.log("delta 2")
      var obj = {"message":"can't insert the data to the database, User collection","error_messages":error};
      res.status(500).json({obj});
      return 0;
    }
  })
});
