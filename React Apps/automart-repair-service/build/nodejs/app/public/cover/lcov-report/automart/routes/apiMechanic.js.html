// add express here
var express = require('express');
var router = express.Router();
 
// add controllers here
var adminController  = require('../controllers/adminController')
var verifyController = require('../controllers/verifyController')
var apiMechanicController  = require('../controllers/apiMechanicController')
 
// add middleware here
var verifyMech = verifyController.verifyAll;
var verifyMechIndiv = verifyController.verifyAllIndiv;
var verifyMech = verifyController.verifyMechanic;
var verifyMechIndiv = verifyController.verifyMechanicIndiv;
 
// add multer here
var multer  = require('multer');
var upload_car = multer({ dest: 'public/uploads/car/images/' });
var upload_mechanic = multer({ dest: 'public/uploads/mechanic/images/' });
 
router.route('/mechanic').get(verifyMech, apiMechanicController.getMechanic)
router.get  ('/mechanic/:id', verifyMech, apiMechanicController.showMechanic)
router.put(   '/mechanic/:id',verifyMechIndiv,    apiMechanicController.putMechanic)
router.put(   '/mechanicImg/:id',upload_mechanic.single('file'),    apiMechanicController.putMechanicImg)
 
router.get('/classDataWithUser/:userType',verifyMech,apiMechanicController.getClassDataWithUser)
router.get('/classDataWithUser/mechanic/:id',verifyMechIndiv,apiMechanicController.showClassDataWithUser)
 
router.route('/part').get(verifyMech, apiMechanicController.getPart)
router.get(   '/part/:id',verifyMech,   apiMechanicController.showPart)
 
router.route('/car').get(verifyMech   ,    apiMechanicController.getCar)
router.get(  '/car/:id',verifyMech,   apiMechanicController.showCar)
 
router.route('/user')
  // .post(verifyAdmin  ,   adminController.postUser)
  .get(verifyMech   ,    apiMechanicController.getUser)
router.get(   '/user/:id',verifyMech,   apiMechanicController.showUser)
router.put(   '/user/:id',verifyMechIndiv,    apiMechanicController.putUser)
 
router.route('/garage').get(verifyMech   ,    apiMechanicController.getGarage)
router.get  ('/garage/:id',verifyMech,   apiMechanicController.showGarage)
 
router.route('/service')
.get(verifyMech   ,    apiMechanicController.getService)
.post(verifyMech  ,   apiMechanicController.postService)
router.get(   '/service/:id',verifyMech,   apiMechanicController.showService)
router.put(   '/service/:id',verifyMech,    apiMechanicController.putService)
 
router.route('/request')
  .get(verifyMech   ,    apiMechanicController.getRequest)
router.get(   '/request/:id',verifyMech,   apiMechanicController.showRequest)
router.put(   '/request/:id',verifyMech,    apiMechanicController.putRequest)
 
router.route('/billing')
.get(verifyMech   ,    apiMechanicController.getBilling)
.post(verifyMech  ,   apiMechanicController.postBilling)
router.get(   '/billing/:id',verifyMech,   apiMechanicController.showBilling)
router.put(   '/billing/:id',verifyMech,    apiMechanicController.putBilling)
 
router.route('/promotion').get(verifyMech   ,    apiMechanicController.getPromotion)
router.get(   '/promotion/:id',verifyMech,   apiMechanicController.showPromotion)
 
module.exports = router;
